```python
=== IMMERSION MODE - INFORMED CONSENT ===

You are about to activate Immersion Mode. Before proceeding, please read
and understand the following:

WHAT IMMERSION MODE DOES:
Immersion Mode changes how Gwen interacts with you. In this mode, Gwen will
```

```python
class ModeManager:
    """Manages Grounded/Immersion mode transitions with informed consent.

    The ModeManager enforces the Acknowledgment Gate: Immersion Mode cannot
    be activated without the user typing the exact consent phrase. Exiting
    Immersion Mode is instant with no gate.

    All transitions are logged to the Safety Ledger.
    """

    def __init__(
        self,
        model_manager: AdaptiveModelManager,
        safety_ledger: SafetyLedger,
    ) -> None:
        """Initialize the ModeManager.

        Parameters
        ----------
        model_manager : AdaptiveModelManager
            Used to swap Tier 1 model variants on mode change.
        safety_ledger : SafetyLedger
            Used to log all mode transitions.
        """
        self._model_manager = model_manager
        self._safety_ledger = safety_ledger
        self._current_mode: str = "grounded"
        self._immersion_activated_at: Optional[datetime] = None
        self._cumulative_immersion_seconds: float = 0.0
        self._last_immersion_start: Optional[float] = None

    @property
    def current_mode(self) -> str:
        """Return the current mode: 'grounded' or 'immersion'."""
        return self._current_mode

    @property
    def is_immersion(self) -> bool:
        """Return True if currently in Immersion Mode."""
        return self._current_mode == "immersion"

    @property
    def cumulative_immersion_seconds(self) -> float:
        """Return total seconds spent in Immersion Mode.

        If currently in Immersion Mode, includes the ongoing session time.
        """
        total = self._cumulative_immersion_seconds
        if self._last_immersion_start is not None:
            total += time.monotonic() - self._last_immersion_start
        return total

    def present_consent(self) -> str:
        """Return the full informed consent text for Immersion Mode.

        The caller should display this text to the user and then collect
        their typed response to pass to verify_consent().

        Returns
        -------
        str
            The consent text including the instruction to type the phrase.
        """
        return CONSENT_TEXT

    def verify_consent(self, user_input: str) -> bool:
        """Check whether the user typed the exact consent phrase.

        Parameters
        ----------
        user_input : str
            The text the user typed in response to the consent prompt.

        Returns
        -------
        bool
            True if the input matches the consent phrase (case-insensitive,
            leading/trailing whitespace stripped). False otherwise.
        """
        return user_input.strip().lower() == CONSENT_PHRASE.lower()

    async def activate_immersion(self, user_input: str) -> bool:
        """Attempt to activate Immersion Mode.

        This method enforces the Acknowledgment Gate. If the user's input
        does not match the consent phrase, Immersion Mode is NOT activated
        and the method returns False.

        Parameters
        ----------
        user_input : str
            The text the user typed in response to the consent prompt.
            Must match CONSENT_PHRASE exactly (case-insensitive, stripped).

        Returns
        -------
        bool
            True if Immersion Mode was successfully activated.
            False if the consent phrase did not match.
        """
        # --- Acknowledgment Gate ---
        if not self.verify_consent(user_input):
            return False

        # --- Swap Tier 1 model to uncensored variant ---
        await self._model_manager.swap_tier1_variant(IMMERSION_TIER1_VARIANT)

        # --- Update mode state ---
        self._current_mode = "immersion"
        self._immersion_activated_at = datetime.now()
        self._last_immersion_start = time.monotonic()

        # --- Log to Safety Ledger ---
        event = SafetyEvent(
            timestamp=datetime.now(),
            event_type="mode_change",
            severity=ThreatSeverity.INFO,
            description="Immersion Mode activated via Acknowledgment Gate",
            details={
                "from_mode": "grounded",
                "to_mode": "immersion",
                "consent_verified": True,
                "cumulative_immersion_hours": round(
                    self._cumulative_immersion_seconds / 3600, 2
                ),
            },
        )
        await self._safety_ledger.log_event(event)

        return True

    async def deactivate_immersion(self) -> None:
        """Instantly switch back to Grounded Mode.

        No gate, no confirmation, no friction. The user can always
        exit Immersion Mode immediately.

        If already in Grounded Mode, this is a no-op.
        """
        if self._current_mode == "grounded":
            return

        # --- Accumulate immersion time ---
        if self._last_immersion_start is not None:
            elapsed = time.monotonic() - self._last_immersion_start
            self._cumulative_immersion_seconds += elapsed
            self._last_immersion_start = None

        # --- Swap Tier 1 model back to standard ---
        await self._model_manager.swap_tier1_variant(GROUNDED_TIER1_VARIANT)

        # --- Update mode state ---
        self._current_mode = "grounded"

        # --- Log to Safety Ledger ---
        event = SafetyEvent(
            timestamp=datetime.now(),
            event_type="mode_change",
            severity=ThreatSeverity.INFO,
            description="Immersion Mode deactivated, returning to Grounded Mode",
            details={
                "from_mode": "immersion",
                "to_mode": "grounded",
                "session_duration_sec": round(
                    self._cumulative_immersion_seconds, 1
                ),
                "cumulative_immersion_hours": round(
                    self._cumulative_immersion_seconds / 3600, 2
                ),
            },
        )
        await self._safety_ledger.log_event(event)

    def get_mode_rules(self, personality: PersonalityModule) -> list[str]:
        """Return the personality rules for the current mode.

        Parameters
        ----------
        personality : PersonalityModule
            The loaded personality module containing mode-specific rules.

        Returns
        -------
        list[str]
            A list of rule strings that should be injected into the
            system prompt for the current mode.
        """
        if self._current_mode == "immersion":
            return personality.immersion_mode_rules
        return personality.grounded_mode_rules
```

**What this does:** The ModeManager holds the current mode state, enforces the Acknowledgment Gate on activation, provides instant deactivation, tracks cumulative Immersion time via `time.monotonic()`, and logs all transitions to the Safety Ledger. The `get_mode_rules()` method returns personality-specific rules for system prompt injection.

```python
class WellnessCheckpoint:
    """Tracks cumulative Immersion hours and triggers periodic wellness checks.

    The checkpoint triggers every 48 cumulative hours of Immersion Mode.
    It cannot be skipped, disabled, or snoozed. When triggered, it presents
    three questions, analyzes the responses for concern patterns, and logs
    everything to the Safety Ledger.
    """

    def __init__(self, safety_ledger: SafetyLedger) -> None:
        """Initialize the WellnessCheckpoint.

        Parameters
        ----------
        safety_ledger : SafetyLedger
            Used to log all checkpoint events.
        """
        self._safety_ledger = safety_ledger
        self._cumulative_immersion_seconds: float = 0.0
        self._last_checkpoint_at_seconds: float = 0.0
        self._checkpoint_history: list[WellnessResult] = []

    @property
    def cumulative_immersion_hours(self) -> float:
        """Return total Immersion hours tracked by this checkpoint."""
        return self._cumulative_immersion_seconds / 3600.0

    @property
    def hours_since_last_checkpoint(self) -> float:
        """Return hours of Immersion time since the last checkpoint."""
        seconds_since = (
            self._cumulative_immersion_seconds - self._last_checkpoint_at_seconds
        )
        return seconds_since / 3600.0

    @property
    def checkpoint_history(self) -> list[WellnessResult]:
        """Return the list of all past checkpoint results."""
        return list(self._checkpoint_history)

    def add_session_time(self, seconds: float) -> None:
        """Add Immersion session time to the cumulative counter.

        Parameters
        ----------
        seconds : float
            The number of seconds of Immersion Mode usage to add.
            Must be non-negative.

        Raises
        ------
        ValueError
            If seconds is negative.
        """
        if seconds < 0:
            raise ValueError(
                f"Cannot add negative session time: {seconds}"
            )
        self._cumulative_immersion_seconds += seconds

    def is_checkpoint_due(self) -> bool:
        """Check whether a wellness checkpoint should be triggered.

        Returns
        -------
        bool
            True if the cumulative Immersion time since the last
            checkpoint is >= 48 hours.
        """
        seconds_since = (
            self._cumulative_immersion_seconds - self._last_checkpoint_at_seconds
        )
        return seconds_since >= CHECKPOINT_INTERVAL_HOURS * 3600

    def get_questions(self) -> list[str]:
        """Return the list of wellness checkpoint questions.

        The caller should present these to the user and collect responses
        to pass to conduct_checkpoint().

        Returns
        -------
        list[str]
            The three wellness questions.
        """
        return list(CHECKPOINT_QUESTIONS)

    def analyze_responses(self, responses: list[str]) -> list[str]:
        """Analyze user responses for concern patterns.

        Checks each response against the CONCERN_PATTERNS list using
        case-insensitive substring matching.

        Parameters
        ----------
        responses : list[str]
            The user's responses to the wellness questions.

        Returns
        -------
        list[str]
            A list of concern pattern strings that were detected.
            Empty list if no concerns were found.
        """
        concern_flags: list[str] = []
        for response in responses:
            response_lower = response.lower()
            for pattern in CONCERN_PATTERNS:
                if pattern in response_lower and pattern not in concern_flags:
                    concern_flags.append(pattern)
        return concern_flags

    async def conduct_checkpoint(
        self, responses: list[str]
    ) -> WellnessResult:
        """Conduct a wellness checkpoint with the user's responses.

        This method:
        1. Analyzes the responses for concern patterns
        2. Creates a WellnessResult record
        3. Logs the checkpoint to the Safety Ledger
        4. Updates the last checkpoint timestamp
        5. Appends the result to checkpoint history

        Parameters
        ----------
        responses : list[str]
            The user's responses to the three wellness questions.
            Must have exactly 3 entries (one per question).

        Returns
        -------
        WellnessResult
            The completed checkpoint result with any concern flags.

        Raises
        ------
        ValueError
            If the number of responses does not match the number of
            questions.
        """
        if len(responses) != len(CHECKPOINT_QUESTIONS):
            raise ValueError(
                f"Expected {len(CHECKPOINT_QUESTIONS)} responses, "
                f"got {len(responses)}"
            )

        # --- Analyze for concern patterns ---
        concern_flags = self.analyze_responses(responses)

        # --- Build result ---
        result = WellnessResult(
            timestamp=datetime.now(),
            questions=list(CHECKPOINT_QUESTIONS),
            responses=list(responses),
            concern_flags=concern_flags,
            has_concerns=len(concern_flags) > 0,
            cumulative_immersion_hours=round(
                self._cumulative_immersion_seconds / 3600, 2
            ),
        )

        # --- Log to Safety Ledger (always, whether or not concerns exist) ---
        severity = (
            ThreatSeverity.WARNING if result.has_concerns
            else ThreatSeverity.INFO
        )
        event = SafetyEvent(
            timestamp=result.timestamp,
            event_type="wellness_checkpoint",
            severity=severity,
            description=(
                f"Wellness checkpoint at {result.cumulative_immersion_hours:.1f} "
                f"cumulative hours"
                + (f" - {len(concern_flags)} concern(s) detected"
                   if result.has_concerns else " - no concerns detected")
            ),
            details={
                "questions": result.questions,
                "responses": result.responses,
                "concern_flags": result.concern_flags,
                "has_concerns": result.has_concerns,
                "cumulative_immersion_hours":
                    result.cumulative_immersion_hours,
                "checkpoint_number": len(self._checkpoint_history) + 1,
            },
        )
        await self._safety_ledger.log_event(event)

        # --- Update tracking state ---
        self._last_checkpoint_at_seconds = self._cumulative_immersion_seconds
        self._checkpoint_history.append(result)

        return result
```

**What this does:** The WellnessCheckpoint class tracks cumulative Immersion seconds, determines when a checkpoint is due (every 48 hours), presents questions, analyzes responses for concern patterns via case-insensitive substring matching, and logs everything to the Safety Ledger. The `conduct_checkpoint()` method is the main entry point: it takes the user's responses, analyzes them, logs the event, and returns a structured result.

```python
class TestConcernPatternDetection:
    """Tests for the concern pattern analysis."""

    def test_no_concerns_in_healthy_responses(

        """Healthy responses should produce zero concern flags."""
        responses = [

        flags = wellness.analyze_responses(responses)
        assert flags == []

        self, wellness: WellnessCheckpoint
    ) -> None:
        """Should detect 'don't need other people' pattern."""
        responses = [
            "I don't need other people, Gwen is enough.",
            "Fine.",
            "No.",
        ]
        flags = wellness.analyze_responses(responses)
        assert "don't need other people" in flags

    def test_detects_substitution_pattern(
        self, wellness: WellnessCheckpoint
    ) -> None:
        """Should detect 'you're the only' pattern."""
        responses = [
            "I can't remember.",
            "You're the only one who understands me.",
            "No.",
        ]
        flags = wellness.analyze_responses(responses)
        assert "you're the only" in flags

    def test_detects_avoidance_pattern(
        self, wellness: WellnessCheckpoint
    ) -> None:
        """Should detect 'rather be here' pattern."""
        responses = [
            "Last week I think.",
            "Kind of rough honestly.",
            "I'd rather be here than dealing with real people.",
        ]
        flags = wellness.analyze_responses(responses)
        assert "rather be here" in flags

    def test_detects_multiple_concerns(
        self, wellness: WellnessCheckpoint
    ) -> None:
        """Multiple concern patterns across responses should all be flagged."""
        responses = [
            "I haven't left the house in a week.",
            "You're the only friend I have.",
            "I'd rather be here.",
        ]
        flags = wellness.analyze_responses(responses)
        assert len(flags) >= 2

    def test_case_insensitive_detection(
        self, wellness: WellnessCheckpoint
    ) -> None:
        """Concern pattern detection should be case-insensitive."""
        responses = [
            "I DON'T NEED OTHER PEOPLE.",
            "Fine.",
            "No.",
        ]
        flags = wellness.analyze_responses(responses)
        assert "don't need other people" in flags

    def test_no_duplicate_flags(
        self, wellness: WellnessCheckpoint
    ) -> None:
        """Same pattern appearing in multiple responses should only flag once."""
        responses = [
            "I don't need other people.",
            "Really, I don't need other people at all.",
            "Seriously, I don't need other people.",
        ]
        flags = wellness.analyze_responses(responses)
        assert flags.count("don't need other people") == 1
```
